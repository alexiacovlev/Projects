OnChange Form Script:

var TaxID= frm.getValue("TaxID");

if (TaxID != "") {

frm.loadInfo("FindCompanyService", TaxID,

  function(res) {

    if (res.Error) alert(res.Error);

    if (res.CompanyID) {

      frm.getElement("SearchInfo_OK").show();

      frm.getElement("SearchInfo_NotFound").hide();

    } else {

      frm.getElement("SearchInfo_OK").hide();

      frm.getElement("SearchInfo_NotFound").show();

    }

 

    frm.setValue("Search_CompanyID", res.CompanyID);

    frm.setValue("CompanyName", res.CompanyName);

    frm.setValue("CompanyNumber", res.CompanyNumber);

    frm.setValue("NatureID", res.NatureID);

    frm.setValue("TypeID", res.TypeID);

  }

);

}

 

Server Side:

\FM\_DATA\OCRS_Company\_forms\info_EnterTIN\FindCompanyService.xml

 

<DbService>

         <Query>

         SELECT CompanyID, CompanyName, CompanyNumber, NatureID, TypeID

         FROM OCRS_Company

         WHERE TaxID = @TaxID

         </Query>

         <Assign>

                   <Field name="TaxID" value="INPUT" />

         </Assign>

</DbService>
